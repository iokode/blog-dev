```cs
public class Person {
    private string givenName;
    private string familyName;

    public string GivenName
    {
        get => this.givenName;
        set => this.givenName = value;
    }

    public string FamilyName
    {
        get => this.familyName;
        set => this.familyName = value;
    }

    public string FullName
    {
        get => GivenName + " " + FamilyName;
    }
}
```
